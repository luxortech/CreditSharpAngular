FormatOptionsJson.Section Method

 

Namespace: LuxorTech.CreditSharp.Format

    

 

The Section method can be used to access and modify the configurations for individual sections within the various CreditSharp Format outputs.  The method call returns an instance of the FormatSection class which can then be used to view or modify available settings.

 

Syntax

public FormatSection Section(string sectionName)

 

Parameters:

 

sectionName

Type: string

Indicates which display section settings to retrieve.  This must be one of the known display sections, as follows:

Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoFormat

  {

      public string CreateJson(CommonData common)

      {

          FormatOptions options = new FormatOptions(FormatLayout.Json);

          options.Json.Partitions = Partitioning.IdMatch;

          options.Json.ReportStyle = JsonStyle.Common01;

 

          this.SetCustomJson(options);

 

          FormatData format = new FormatData(options);

          return format.Generate(common);

      }

 

      public void SetCustomJson(FormatOptions options)

      {

          // Disable (turn off) display of three sections

          options.Json.Section("Employment").Visible = false;

          options.Json.Section("Message").Visible = false;

          options.Json.Section("Directory").Visible = false;

      }

  }

}
