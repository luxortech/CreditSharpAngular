MergeData Class

 

Namespace: LuxorTech.CreditSharp.Datamart.Merge

    

 

Root class used to obtain a deduped image from one or more normalized bureau inputs.

Example 1

 

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Datamart.Merge;

 

namespace HelpDemos

{

  public class DemoMerge

  {

      public MergeData MergeBureauData(CommonData input)

      {

          MergeData merge = new MergeData(new MergeOptions());

          merge.Execute(input);

 

          return merge;

      }

  }

}

 

Example 2. Setting option properties

 

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Datamart.Merge;

 

namespace HelpDemos

{

    public class DemoMerge

   {

      public MergeData MergeBureauData(CommonData input, CommonData input2, CommonData input3)

      {

          MergeOptions options = new MergeOptions();

          // Allow primary and secondary data to be merged

          options.AllowJoint = true;

          // Elect to use the Basic compare profile within bureaus and Standard between bureaus

          options.CompareProfile = "Basic+Standard";

          // Elect to use the Pick combine profile for all instances

          options.CombineProfile = "Pick";

 

          MergeData merge = new MergeData(options);

          merge.Execute(input, input2, input3);

 

          return merge;

      }

   }

}
