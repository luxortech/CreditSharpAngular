Compute

 

Namespace: LuxorTech.CreditSharp.Analyze

    

 

A Compute operation is used to calculate a value using one or more results from a previously calculated attribute.  They are useful in making analysis more efficient by reusing the outcome of a single calculation in more than one equation without having to recalculate it as well as for performing meta-analysis using previously calculated aggregates.  

 

The computation is defined using the AddEvaluation method within the AnalysisBundle class and the previously calculated attribute references are accessed using the Item method within the AnalyzeResults class.  The Item method returns an instance of an the desired AnalyzeAttribute class.

 

One example of where Compute operations can be useful is in calculating what ratio different types of tradelines appear in a credit bureau response.  The following statements outline how to arrive at relative percentages of each type of tradeline.  The ComputePredicate statement is used to ensure there is no attempt to divide by zero.

 

          TradelinePredicate isRev = (trade, options) => trade.IsRevolving;

          TradelinePredicate isIns = (trade, options) => trade.IsInstallment;

          TradelinePredicate isMtg = (trade, options) => trade.IsMortgage;

          TradelinePredicate isCrl = (trade, options) => trade.IsCreditLine;

          TradelinePredicate notColl = (trade, options) => !trade.IsCollection;

 

          bundle.AddEvaluation(new EvaluateTradeline("NbrRev", AnalysisOperation.Count, ComTradelineIds.IdCount, isRev));

          bundle.AddEvaluation(new EvaluateTradeline("NbrIns", AnalysisOperation.Count, ComTradelineIds.IdCount, isIns));

          bundle.AddEvaluation(new EvaluateTradeline("NbrMtg", AnalysisOperation.Count, ComTradelineIds.IdCount, isMtg));

          bundle.AddEvaluation(new EvaluateTradeline("NbrCrl", AnalysisOperation.Count, ComTradelineIds.IdCount, isCrl));

          bundle.AddEvaluation(new EvaluateTradeline("NbrTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, notColl));

 

          ComputePredicate haveTrl = (results) => results.Item("NbrTrl").Quantity > 0;

 

          ComputeSolution pctRev = (results) => results.Item("NbrRev").Quantity == 0 ? 0 :

                                   100 * (results.Item("NbrRev").Quantity / results.Item("NbrTrl").Quantity);

 

          ComputeSolution pctIns = (results) => results.Item("NbrIns").Quantity == 0 ? 0 :

                                   100 * (results.Item("NbrIns").Quantity / results.Item("NbrTrl").Quantity);

 

          ComputeSolution pctMtg = (results) => results.Item("NbrMtg").Quantity == 0 ? 0 :

                                   100 * (results.Item("NbrMtg").Quantity / results.Item("NbrTrl").Quantity);

 

          ComputeSolution pctCrl = (results) => results.Item("NbrCrl").Quantity == 0 ? 0 :

                                   100 * (results.Item("NbrCrl").Quantity / results.Item("NbrTrl").Quantity);

 

          bundle.AddEvaluation(new EvaluateReference("PctRev", AnalysisOperation.Compute, haveTrl, pctRev));

          bundle.AddEvaluation(new EvaluateReference("PctIns", AnalysisOperation.Compute, haveTrl, pctIns));

          bundle.AddEvaluation(new EvaluateReference("PctMtg", AnalysisOperation.Compute, haveTrl, pctMtg));

          bundle.AddEvaluation(new EvaluateReference("PctCrl", AnalysisOperation.Compute, haveTrl, pctCrl));
