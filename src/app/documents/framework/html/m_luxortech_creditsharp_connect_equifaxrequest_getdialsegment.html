EquifaxRequest.GetDialSegment Method

 

Namespace: LuxorTech.CreditSharp.Connect

    

 

The GetDialSegment method is used to extract the string which would be submitted to Equifax as part of the inquiry given the current EquifaxRequest settings.  This output is typically paired with the return value from the GetTextInquiry method to provide a full picture of the Equifax text based inquiry.

 

Syntax

public string GetDialSegment ()

 

Parameters: none

 

Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Connect;

 

namespace HelpDemos

{

  public class GetEfxInquiryData

  {

      public string GetInquiryData(string outputType, Applicant primary, Applicant secondary)

      {

          BureauConnection connection = new BureauConnection(BureauIds.Equifax);

          connection.HttpSettings.ServiceName = "acrotest";

          connection.HttpSettings.ServiceURL = "https://transport5.ec.equifax.com/servlet/stspost";

          connection.HttpSettings.WebUserId = "myCompanyId";

          connection.HttpSettings.WebPassword = "letMeIn";

 

          EfxInquiryOptions options = new EfxInquiryOptions(licenseKey);

          options.ConsumerReferral = true;

          options.MilitaryBorrowerCheck = true;

 

          EquifaxRequest request = new EquifaxRequest(options);

          request.PermissiblePurposeCode = "01";

          request.ProductCode = EfxProductCode.CreditReport;

          request.ResponseVersion = EfxVersionNumber.Version50;

          request.SubscriberCode = "222XX11111";

          request.SubscriberPassword = "ABC";

 

          if (secondary != null)

               request.IsJointRequest = true;

 

          string output = string.Empty;

          switch (outputType)

          {

              case "full":

                  output = request.GetInquiry(connection, primary, secondary);

                  break;

              case "dial":

                  output = request.GetDialSegment();

                  break;

              case "text":

                  output = request.GetTextInquiry(primary, secondary);

                  break;

              case "xml":

                  output = request.GetXmlInquiry(primary, secondary);

                  break;

          }

 

          return output;

      }

   }

}
