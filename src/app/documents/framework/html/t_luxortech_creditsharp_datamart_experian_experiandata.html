Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoCommon

  {

      public string GetXmlOutput(string rawCreditInput, BureauIds bureauId)

      {

          BureauData bureauData = null;

          switch (bureauId)

          {

              case BureauIds.Equifax:

                  bureauData = new EquifaxData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.Experian:

                  bureauData = new ExperianData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.TransUnion:

                  bureauData = new TransUnionData(new BureauOptions(licenseKey));

                  break;

          }

          bureauData.Parse(rawCreditInput);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(bureauData);

 

          FormatOptions options = new FormatOptions(FormatLayout.Xml);

          options.Xml.Partitions = Partitioning.IdMatch;

          FormatData formatData = new FormatData(options);

 

          return formatData.Generate(commonData);

      }

   }

}

AddModel

CreateArf

Creates an Automated Response Format (machine readable) string using whatever bureau data that was imported to the current object.

InsertJsonHeader

Injects header information (date credit was obtained and a user defined reference number, if desired) into an existing JSON formatted credit bureau response

 

Experian does not currently return the date ordered within their JSON responses. Since that value is vital when performing date-based arithmetic during credit analysis, this method can be used to supply that detail.

 

Note, this is only required if the credit data was obtained outside of the CreditSharp framework. If credit is requested using CreditSharp, then this header information is inserted into the JSON bureau response at the time the inquiry is made.

InsertJsonRefNbr

Adds a reference number that can be included in various CreditSharp output formats.

 

Unlike with NetConnect, APIHUB JSON transactions with Experian currently do not have the ability to echo back a user defined reference number in their JSON responses.

 

This method can be used to inject that data item so that it can then be included in various CreditSharp outputs.

Parse

Populates an instance of the ExperianData class by extracting individual bureau elements from the input buffer.

DecisionSchema

Text

Gets or sets the value to be used when parsing an Experian decisioning response resulting from an Instant Prescreen or Prequalification request.

 

The content is a small schema presented in JSON format which describes the custom data elements that can be expected within the decision response.  These definitions are used to populate individual instances of ExpBureauAttribute segments thereby allowing for easy extraction of decisioning results.

 

For the custom extraction process to be successful, the schema must be assigned prior to invoking the Parse method.

 

For further discussion, please refer to the Ordering Custom Solutions topic in the Experian Connect chapter and Processing Custom Solutions later in this chapter.

FilteredView

Bool

Gets or sets the property indicating if data elements will be presented in their cleansed or raw formats.

 

When true, elements will be displayed in consistent, type appropriate formats.  Amounts and numbers will be true numeric values with leading and trailing spaces trimmed and any leading zeroes removed. Character based fields will likewise have all leading and trailing spaces removed.  Date fields will conform to whatever DateMask option was chosen.

 

When false, all elements will be displayed in whatever raw form was provided by the bureau.

 

[Defaults to true]
