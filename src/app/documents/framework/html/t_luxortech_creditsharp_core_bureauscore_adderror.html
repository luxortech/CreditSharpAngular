AddError Method

 

    

 

The AddError method is used to append a look up value for an error code and is textual translation for a custom bureau scoring model.

 

Syntax

public void AddError (string errorCode, string errorText)

 

Parameters:

 

errorCode

Type: string

The code associated with the particular error

errorText

Type: string

The textual translation which corresponds to the current code

 

AddError Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoCustom

  {

        public string EquifaxCustom(string creditDataBuffer)

        {

            EquifaxData equifaxData = new EquifaxData(new BureauOptions("licenseKey"));

            BureauModel customModel = equifaxData.AddModel("CM1", "Custom Model One");

            customModel.AddFactor("01", "Adverse factor code number one");

            customModel.AddFactor("02", "Adverse factor code number two");

            customModel.AddFactor("03", "Adverse factor code number three");

            customModel.AddFactor("04", "Adverse factor code number four");

 

            customModel.AddError("4", "Error number four occurred while computing score.");

 

            equifaxData.Parse(creditDataBuffer);

 

            CommonData commonData = new CommonData(new CommonOptions());

            commonData.Normalize(equifaxData);

 

            FormatData formatData = new FormatData(new FormatOtions(FormatLayout.Html));

            return formatData.Generate(commonData);

        }

  }

}
