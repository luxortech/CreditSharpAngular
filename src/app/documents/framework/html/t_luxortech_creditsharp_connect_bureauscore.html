BureauScore Class

 

Namespace: LuxorTech.CreditSharp.Connect

    

 

An instance of the BureauScore class is returned by a call to one of the bureau specific request object's AddModel methods (EquifaxRequest, ExperianRequest or TransUnionRequest).  It serves as a reference point for assigning any additional parameters beyond simply asking that the score model be included in the bureau response.

 

Example

 

      public string OrderBureau(Applicant primary, Applicant secondary)

      {

          BureauConnection connection = new BureauConnection(BureauIds.TransUnion);

          connection.HttpSettings.ServiceURL = "https://netaccess-test.transunion.com";

          connection.HttpSettings.WebUserId = "myCompanyId";

          connection.HttpSettings.WebPassword = "notPassword";

 

          connection.HttpSettings.CertificateStore = "Trusted Root";

          connection.HttpSettings.CertificateLocation = "CurrentUser";

          connection.HttpSettings.CertificateName = "TUNA Cert";

 

          TucInquiryOptions options = new TucInquiryOptions(licenseKey);

          options.RequestRegion = TucRequestRegion.ProductionSystem;

 

          TransUnionRequest request = new TransUnionRequest(options);

          request.PermissiblePurposeCode = "AU";

          request.ProductCode = TucProductCode.CreditReport;

          request.ResponseVersion = TucVersionNumber.Version40;

          request.SubscriberCode = "1234567";

          request.SubscriberPassword = "A999";

          request.SubscriberMarket = "06";

          request.SubscriberSubMarket = "22";

          request.TtyResponse = TucTtyResponse.ContinuousSheet;

 

          BureauScore bureauScore = request.AddModel("00R82");

          bureauScore.SupplementalData = ScoreSupplement.RangeAndRank;

 

          string creditData = string.Empty;

          if (secondary != null)

                creditData = request.RequestProduct(connection, primary, secondary);

          else

               creditData = request.RequestProduct(connection, primary);

 

          return creditData;

      }
