GetJsonInquiry Method

 

Namespace: LuxorTech.CreditSharp.Connect

    

 

The GetJsonInquiry method is used to extract the string which would be submitted to Experian given the current BureauConnection and Applicant data and a ConnectionType of ConnectionType.Http.

 

Method Overloads

 

Name

Description

GetJsonInquiry(BureauConnection connection, Applicant primary)

Returns an inquiry string for an individual request.

GetJsonInquiry(BureauConnection connection, Applicant primary, Applicant secondary)

Returns an inquiry string for a joint request.

 

Syntax

public string GetJsonInquiry (BureauConnection connection, Applicant primary)

public string GetJsonInquiry (BureauConnection connection, Applicant primary, Applicant secondary)

 

Parameters:

 

connection

Type: BureauConnection object reference

Contains a populated bureau connection object that would be used to submit a inquiry to the bureau.

primary

Type: Applicant object reference

Contains all the personally identifiable information for the primary applicant.

 

secondary

Type: Applicant object reference

Contains all the personally identifiable information for the secondary applicant.

 

Note: the IsJointRequest property in the ExperianRequest object must still be set to true in order for a joint request to be submitted or created.

 

Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Connect;

 

namespace HelpDemos

{

  public class GetExpInquiryData

   {

      public string GetInquiryData(string outputType, Applicant primary, Applicant secondary)

       {

          BureauConnection connection = new BureauConnection(BureauIds.Experian);

             connection.HttpSettings.AuthorizationId = "myAuthId";

             connection.HttpSettings.AuthorizationSecret = "myAuthSecret";

             connection.HttpSettings.ServiceName = "API";

             connection.HttpSettings.ServiceURL = "https://uat-us-api.experian.com/oauth2/v1/token";

             connection.HttpSettings.WebUserId = "myCompanyId";

           connection.HttpSettings.WebPassword = "notPassword";

 

          ExperianRequest request = new ExperianRequest(new ExpInquiryOptions(licenseKey));

           request.PermissiblePurposeCode = "98";

           request.ProductCode = ExpProductCode.CreditReport;

           request.ResponseVersion = ExpVersionNumber.Version20;

           request.SubscriberPreamble = "TEST";

           request.SubscriberCode = "1234567";

           request.SubscriberPassword = "ABC";

           request.JsonResponse = ExpJsonResponse.Standard;

 

          if (secondary != null)

               request.IsJointRequest = true;

 

          string output = string.Empty;

          switch (outputType)

           {

              case "full":

                   output = request.GetInquiry(connection, primary, secondary);

                  break;

              case "text":

                  // return the TCP/IP inquiry string

                   output = request.GetTextInquiry(primary, secondary);

                  break;

              case "xml":

                  // return the NetConnect inquiry string

                  output = request.GetXmlInquiry(primary, secondary);

                  break;

              case "json":

                  // return the APIHUB inquiry string

                  output = request.GetJsonInquiry(primary, secondary);

                  break;

           }

 

          return output;

       }

   }

}
