Exception Handling

 

    

 

All exceptions that may be generated from within the CreditSharp Framework are derived from the same base class: CreditSharpException.  As such, error handling can be centralized to catch just a single CreditSharpException, or the error handling can be narrowed to specific namespaces.  An example of each approach is presented below.

 

With regard to ParameterException handling, each component within the Framework can potentially throw a parameter exception.  These exceptions include additional properties to indicate the specific data element that was found to be invalid (ElementName) and usually, the offending value (ElementValue).

 

Each error instance will also typically include a message further describing the error condition, its Message property.  Any operating system errors that may occur during Framework execution will also be trapped and wrapped in a CreditSharpException.

 

Exception for all

 

using System;

using System.Diagnostics;

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoConsolidatedException

  {

      public string GetHtmlReport(BureauIds bureauId, Applicant primary, Applicant secondary)

      {

          string output = string.Empty;

          OrderBureau orderBureau = new OrderBureau();

          string creditData = orderBureau.SubmitRequest(bureauId, primary, secondary);

 

          if (creditData == "Request failed.")

              return "Request failed.";

 

          BureauData bureauData = null;

          try

          {

              switch (bureauId)

              {

                  case BureauIds.Equifax:

                      bureauData = new EquifaxData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.Experian:

                      bureauData = new ExperianData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.TransUnion:

                      bureauData = new TransUnionData(new BureauOptions(licenseKey));

                      break;

              }

 

              bureauData.Parse(creditData);

 

              CommonData commonData = new CommonData(new CommonOptions());

              commonData.Normalize(bureauData);

 

              FormatData formatData = new FormatData(new FormatOptions(FormatLayout.Html));

              output = formatData.Generate(commonData);

          }

 

          catch (CreditSharpException ex)

          {

              Debug.WriteLine("An error while processing the request for {0}",

                                  primary.Identity.FirstName + " " + primary.Identity.LastName);

              Debug.WriteLine(ex.Message);

              output = "Request failed.";

          }

 

          return output;

      }

   }

}

                  case BureauIds.Equifax:

                      bureauData = new EquifaxData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.Experian:

                      bureauData = new ExperianData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.TransUnion:

              output = formatData.Generate(commonData);

          }

 

          catch (CreditSharpException ex)

          {

              if (ex is BureauDataException)

              {

                  Debug.WriteLine("An error occurred importing bureau data on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is NormalizeDataException)

              {

                  Debug.WriteLine("An error occurred normalizing bureau data on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is FormatDataException)

              {

                  Debug.WriteLine("An error occurred generating report on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is ParameterException)

              {

                  Debug.WriteLine("An input paremter is incorrect.");

              }

              else

              {

                  Debug.WriteLine("An error while processing the request for {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              Debug.WriteLine(ex.Message);

              output = "Request failed.";

          }

 

          return output;

      }

   }

}

using System;

using System.Diagnostics;

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoExceptionByTask

  {

      public string GetHtmlReport(BureauIds bureauId, Applicant primary, Applicant secondary)

      {

          string output = string.Empty;

          OrderBureau orderBureau = new OrderBureau();

          string creditData = orderBureau.SubmitRequest(bureauId, primary, secondary);

 

          if (creditData == "Request failed.")

              return "Request failed.";

 

          BureauData bureauData = null;

          CommonData commonData = null;

 

          try

          {

              switch (bureauId)

              {

                  case BureauIds.Equifax:

                      bureauData = new EquifaxData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.Experian:

                      bureauData = new ExperianData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.TransUnion:

                      bureauData = new TransUnionData(new BureauOptions(licenseKey));

                      break;

              }

 

              bureauData.Parse(creditData);

          }

 

          catch (ParameterException ex)

          {

              Debug.WriteLine("An invalid parameter was sent to the BureauData class: Name '{0}' Value'{1}'", ex.ElementName, ex.ElementValue);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          catch (BureauDataException ex)

          {

              Debug.WriteLine("An error occurred importing bureau data on {0}",

                                  primary.Identity.FirstName + " " + primary.Identity.LastName);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          try

          {

              commonData = new CommonData(new CommonOptions());

              commonData.Normalize(bureauData);

          }

 

          catch (ParameterException ex)

          {

              Debug.WriteLine("An invalid parameter was sent to the CommonData class: Name '{0}' Value'{1}'", ex.ElementName, ex.ElementValue);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          catch (NormalizeDataException ex)

          {

              Debug.WriteLine("An error occurred normalizing bureau data on {0}",

                                  primary.Identity.FirstName + " " + primary.Identity.LastName);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          try

          {

              FormatData formatData = new FormatData(new FormatOptions(FormatLayout.Html));

              output = formatData.Generate(commonData);

          }

 

          catch (FormatDataException ex)

          {

              Debug.WriteLine("An error occurred generating report on {0}",

                                  primary.Identity.FirstName + " " + primary.Identity.LastName);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          catch (ParameterException ex)

          {

              Debug.WriteLine("An invalid parameter was sent to the FormatData class: Name '{0}' Value'{1}'", ex.ElementName, ex.ElementValue);

              Debug.WriteLine(ex.Message);

              return "Request failed.";

          }

 

          return output;

      }

   }

}
