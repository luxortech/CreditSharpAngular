ExtractData Class

 

Namespace: LuxorTech.CreditSharp.Extract

    

 

The ExtractData class is the root class from which native bureau data can be retrieved from the CreditSharp framework.

A typical programming pattern

 

using System.Diagnostics;

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Extract;

 

namespace HelpDemos

{

  public class DemoExtract

  {

      public string GetBureauXml(string creditDataBuffer, BureauIds creditBureau, string licenseKey)

      {

          BureauData bureauData = null;

          string output = string.Empty;

 

          switch(creditBureau)

          {

              case BureauIds.Equifax:

                  bureauData = new EquifaxData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.Experian:

                  bureauData = new ExperianData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.TransUnion:

                  bureauData = new TransUnionData(new BureauOptions(licenseKey));

                  break;

          }

 

          try

          {

              bureauData.Parse(creditDataBuffer);

 

              ExtractData extractData = new ExtractData(new ExtractOptions(ExtractLayout.Xml));

              output = extractData.Assemble(bureauData);

          }

 

          catch (CreditSharpException ex)

          {

              Debug.WriteLine(LuxorTech.CreditSharp.Core.Utilities.ErrorStack(ex));

          }

 

          return output;

      }

   }

}
