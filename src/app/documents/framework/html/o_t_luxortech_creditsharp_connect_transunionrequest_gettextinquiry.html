TransUnionRequest.GetTextInquiry Method

 

Namespace: LuxorTech.CreditSharp.Connect

    

 

The GetTextInquiry method is used to extract the string which would be submitted to TransUnion given the current BureauConnection and Applicant data when the requested layout is one of the TucCpuResponse enumerations.

 

Method Overloads

 

Name

Description

GetInquiry(BureauConnection connection, Applicant primary)

Returns an inquiry string for an individual request.

GetInquiry(BureauConnection connection, Applicant primary, Applicant secondary)

Returns an inquiry string for a joint request.

 

Syntax

public string GetInquiry (BureauConnection connection, Applicant primary)

public string GetInquiry (BureauConnection connection, Applicant primary, Applicant secondary)

 

Parameters:

 

connection

Type: BureauConnection object reference

Contains a populated bureau connection object that would be used to submit a inquiry to the bureau.

primary

Type: Applicant object reference

Contains all the personally identifiable information for the primary applicant.

 

secondary

Type: Applicant object reference

Contains all the personally identifiable information for the secondary applicant.

 

Note: the IsJointRequest property in the TransUnionRequest object must still be set to true in order for a joint request to be submitted or created.

 

Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Connect;

 

namespace HelpDemos

{

  public class TransUnionConnect

   {

      public string GetInquiryData(string outputType, Applicant primary, Applicant secondary)

       {

          BureauConnection connection = new BureauConnection(BureauIds.TransUnion);

           connection.HttpSettings.ServiceURL = "https://netaccess-test.transunion.com";

           connection.HttpSettings.WebUserId = "myCompanyId";

           connection.HttpSettings.WebPassword = "notPassword";

 

           connection.HttpSettings.CertificateStore = "Trusted Root";

           connection.HttpSettings.CertificateLocation = "CurrentUser";

           connection.HttpSettings.CertificateName = "TUNA Cert";

 

          TucInquiryOptions options = new TucInquiryOptions(licenseKey);

           options.RequestRegion = TucRequestRegion.ProductionSystem;

 

          TransUnionRequest request = new TransUnionRequest(options);

           request.PermissiblePurposeCode = "AU";

           request.ProductCode = TucProductCode.CreditReport;

           request.ResponseVersion = TucVersionNumber.Version40;

           request.SubscriberCode = "1234567";

           request.SubscriberPassword = "A999";

           request.SubscriberMarket = "06";

           request.SubscriberSubMarket = "22";

           request.TtyResponse = TucTtyResponse.ContinuousSheet;

 

          if (secondary != null)

               request.IsJointRequest = true;

 

          string output = string.Empty;

          switch (outputType)

           {

              case "full":

                   output = request.GetInquiry(connection, primary, secondary);

                  break;

              case "text":

                  // return the TU4I inquiry string

                  output = request.GetTextInquiry(primary, secondary);

                  break;

              case "xml":

                  // return the TUXML inquiry string

                   output = request.GetXmlInquiry(primary, secondary);

                  break;

           }

 

          return output;

       }

}
