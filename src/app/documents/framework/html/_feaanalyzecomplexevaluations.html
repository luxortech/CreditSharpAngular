Constructing complex evaluations

 

    

 

The following examples illustrate two somewhat complex computations.  One example can be used to compute the percentage of available credit a consumer has currently accessed on their revolving accounts.  The second example computes the average number of months all non-collection accounts are/were open.

 

The first example uses inline lambda expressions to define the custom evaluation while the second references functions that will be invoked when the analysis is executed.

 

Utilization

      /// <summary>

      /// Instructions for defining and adding custom Reference (Compute) attributes using inline lambda references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InlinedReference(AnalysisBundle bundle)

      {

          //

          // This example illustrates how to use internal attribute references to compute a value based upon other aggregates.

          // Here, a revolving utilization percentage is computed.

          //

 

          //

          // First define global references to be used for computing Revolving Utilization. Make sure we only include tradelines that are:

          //      A) Revolving, and

          //      B) have a valid CreditLimit or HighCredit

          //

          // The special '@' character indicates the attribute value is to be used for internal calculations only and will not be

          // included in any outputs. Reference attributes can include any aggregated value, not just these internal only values.

          //

          // The AMT functions are passed a local assumeZero parameter so we can be sure of our computation

          // regardless of what the global AssumeZero property contains

          //

          TradelinePredicate isRevWithAmts = (trade, options) => trade.IsRevolving &&

                                                                (

                                                                 trade.AMT(ComTradelineIds.CreditLimitAmount, true) > 0 ||

                                                                 trade.AMT(ComTradelineIds.HighCreditAmount, true) > 0

                                                                );

 

          // Return whichever value is non-zero, favoring CreditLimit over HighCredit

          TradelineSolution utlizationAmt = (trade, options) => trade.EMPTY(ComTradelineIds.CreditLimitAmount) ?

                                                                trade.AMT(ComTradelineIds.HighCreditAmount) :

                                                                trade.AMT(ComTradelineIds.CreditLimitAmount);

 

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevBalances", AnalysisOperation.Sum, ComTradelineIds.BalanceAmount, isRevWithAmts));

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevLimits", AnalysisOperation.Sum, ComTradelineIds.IdCount, isRevWithAmts, utlizationAmt));

 

          //

          // Now, construct the utilization computation using attribute references.  

          // If the Balance is zero, utilization must be zero, otherwise compute the percentage

          //

          bundle.AddEvaluation(new EvaluateReference("TRL_Utilization", AnalysisOperation.Compute,

          (results) => results.Item("@TRL_RevBalances").Quantity == 0 ? 0 :

          100 * (results.Item("@TRL_RevBalances").Quantity / results.Item("@TRL_RevLimits").Quantity)));

      }

 

Months Open

      public void MonthsOpen(AnalysisBundle bundle)

      {

          //

          // Add the new evaluation, which includes references the functions defined below that will be invoked during analysis execution

          //

          bundle.AddEvaluation(new EvaluateTradeline("TRL_AvgMonthsOpen",

                                AnalysisOperation.Average, ComTradelineIds.IdCount,

                               (TradelinePredicate)TRL_HasOpenNotColl, (TradelineSolution)TRL_MonthsOpen));

      }

 

      private static bool TRL_HasOpenNotColl(ComTradeline trade, AnalyzeOptions options)

      {

          if (trade.EMPTY(ComTradelineIds.DateOpen) == true || trade.IsCollection)

              return false;

          else

              return true;

      }

 

      private static double TRL_MonthsOpen(ComTradeline trade, AnalyzeOptions options)

      {

          if (!trade.EMPTY(ComTradelineIds.DateClosed))

              return trade.DATE(ComTradelineIds.DateClosed) - trade.DATE(ComTradelineIds.DateOpen);

          else if (trade.IsFixedPayment == true)

          {

              //

              // For mortgages and installements, if there is a zero balance we can assume the account is closed

              // and attempt to use a proxy for an explicit closed date

              //

              if (trade.AMT(ComTradelineIds.BalanceAmount) == 0)

              {

                  if (!trade.EMPTY(ComTradelineIds.DatePaid))

                      return trade.DATE(ComTradelineIds.DatePaid) - trade.DATE(ComTradelineIds.DateOpen);

                  else if (!trade.EMPTY(ComTradelineIds.DateLastActive))

                      return trade.DATE(ComTradelineIds.DateLastActive) - trade.DATE(ComTradelineIds.DateOpen);

                  else

                      return trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen);

              }

              else

                  return trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen);

          }

          else

              //

              // We have an open reolving account of sort and can only compute an age using the most recent report date

              //

              return trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen);

      }
