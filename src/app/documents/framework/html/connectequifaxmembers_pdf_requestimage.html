Requesting a PDF image from Equifax

 

    

 

Equifax's new PDF reports are requested by enabling a new EfxTtyResponse enumeration when submitting a credit bureau inquiry.  The JSON response will then include a URL (or two URLs if submitting a Joint bureau request) which can then be used to subsequently obtain the actual PDF content.

 

If desired, the CreditSharp EquifaxRequest object can be instructed to automatically download the PDF content, using the URL details provided in the response, and have this data embedded within the native JSON credit bureau envelope.  Doing so will lead to CreditSharp inserting a pdfImage1 (and pdfImage2, if Joint) element into the JSON response which contains the pdf content encoded as base64 data.

 

In the coding example the follows, the RequestWithPDF method demonstrates submitting an inquiry that will include the PDF links in the response data.  The RequestWithEmbeddedPDF method demonstrates enabling the option to have CreditSharp download the PDF content and add it to the response data.

 

Setting up and submitting the request.

 

namespace EquifaxPDFExtract

{

    using LuxorTech.CreditSharp.Core;

    using LuxorTech.CreditSharp.Connect;

 

    public class RequestEquifaxData

    {

        //

        // Equifax credentials

        //

        public string MyAuthorizationId;

        public string MyAuthorizationSecret;

        public string MySubscriberCode;

        public string MySubscriberPswd;

        public string MyCustomerCode;

 

        public string RequestWithPDF(string licenseKey, Applicant subject, Applicant spouse)

        {

            BureauConnection connection = CreateConnection();

            EfxInquiryOptions options = new EfxInquiryOptions(licenseKey);

            return CreateRequest(options).RequestProduct(connection, subject, spouse);

        }

 

        public string RequestWithEmbeddedPDF(string licenseKey, Applicant subject, Applicant spouse)

        {

            BureauConnection connection = CreateConnection();

            //

            // Create options with AppendPdfImage enabled so CreditSharp retrieves the PDF contents and stuffs it into the response

            //

            EfxInquiryOptions options = new EfxInquiryOptions(licenseKey)

            {

                AppendPdfImage = true

            };

            return CreateRequest(options).RequestProduct(connection, subject, spouse);

        }

 

        private BureauConnection CreateConnection()

        {

            BureauConnection connection = new BureauConnection(BureauIds.Equifax);

            connection.HttpSettings.ServiceURL = @"https://api.uat.equifax.com/v2/oauth/token";

            connection.HttpSettings.ServiceName = "api";

            connection.HttpSettings.AuthorizationId = this.MyAuthorizationId;

            connection.HttpSettings.AuthorizationSecret = this.MyAuthorizationSecret;

 

            return connection;

        }

 

        private EquifaxRequest CreateRequest(EfxInquiryOptions options)

        {

            EquifaxRequest request = new EquifaxRequest(options);

            request.SubscriberCode = this.MySubscriberCode;

            request.SubscriberPassword = this.MySubscriberPswd;

            request.CustomerCode = this.MyCustomerCode;

 

            request.ProductCode = EfxProductCode.CreditReport;

            request.ResponseVersion = EfxVersionNumber.Version10;

            request.JsonResponse = EfxJsonResponse.Standard;

            //

            // Include the trigger telling Equifax to generate PDF images and return the URLs

            //

            request.TtyResponse = EfxTtyResponse.PdfImage;

 

            return request;

        }

    }

}
