ExperianRequest Class

 

Namespace: LuxorTech.CreditSharp.Connect

    

 

The ExperianRequest class is used to prepare and execute an Experian request for credit data.  This class is used to transmit requests to the Experian system and capture the responses for later reference.  In addition to transmitting the request, the class also allows for the retrieval the actual inquiry string to aid in record keeping or troubleshooting.

 

The AddModel method is where requests for any bureau scoring models would be configured.  The AddService method can be used to add optional bureau feature requests or products that may require additional inputs or are not already accounted for with dedicated ExpInquiryOptions properties.

NetConnect Example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Connect;

 

    public class ExperianConnect

    {

        public string OrderBureau(Applicant primary, Applicant secondary)

        {

            BureauConnection connection = new BureauConnection(BureauIds.Experian);

           connection.HttpSettings.ServiceName = "STAR";

           connection.HttpSettings.ServiceURL = "http://www.experian.com/lookupServlet1?lookupServiceName=AccessPoint&lookupServiceVersion=1.0&serviceName=NetConnectDemo&serviceVersion=2.0&responseType=text/plain";

           connection.HttpSettings.WebUserId = "myCompanyId";

           connection.HttpSettings.WebPassword = "notPassword";

 

            ExpInquiryOptions options = new ExpInquiryOptions(licenseKey);

           options.DirectCheck = true;

           options.MilitaryBorrowerCheck = true;

           options.Year2000 = true;

 

            ExperianRequest request = new ExperianRequest(options);

           request.PermissiblePurposeCode = "01";

           request.ProductCode = ExpProductCode.CreditReport;

           request.ResponseVersion = ExpVersionNumber.Version70;

           request.SubscriberPreamble = "TEST";

           request.SubscriberCode = "1234567";

           request.SubscriberPassword = "ABC";

           request.XmlResponse = ExpXmlResponse.Standard;

 

            string creditData = string.Empty;

            if (secondary != null)

                creditData = request.RequestProduct(connection, primary, secondary);

            else

               creditData = request.RequestProduct(connection, primary);

 

            return creditData;

        }

   }

 

APIHUB Example

 

  public class ExperianConnect

   {

      public string GetJsonData(string licenseKey, string outputType, Applicant primary, Applicant secondary)

       {

          BureauConnection connection = new BureauConnection(BureauIds.Experian);

           connection.HttpSettings.AuthorizationId = "myAuthId";

           connection.HttpSettings.AuthorizationSecret = "myAuthSecret";

           connection.HttpSettings.ServiceName = "API";

           connection.HttpSettings.ServiceURL = "https://uat-us-api.experian.com/oauth2/v1/token";

           connection.HttpSettings.WebUserId = "myCompanyId";

           connection.HttpSettings.WebPassword = "notPassword";

 

          ExpInquiryOptions options = new ExpInquiryOptions(licenseKey);

           options.ServiceRegion = ExpServiceRegion.Test;

 

          ExperianRequest request = new ExperianRequest(options);

           request.PermissiblePurposeCode = "98";

           request.ProductCode = ExpProductCode.CreditReport;

           request.JsonResponse = ExpJsonResponse.Standard;

           request.ResponseVersion = ExpVersionNumber.Version20;

           request.SubscriberPreamble = "TEST";

           request.SubscriberCode = "1234567";

           request.SubscriberPassword = "ABC";

 

          if (secondary != null)

               request.IsJointRequest = true;

 

          return request.RequestProduct(connection, primary, secondary);

       }

ContractAmount

Num (3)

Amount of credit contract (ie: loan amount) that would apply given a credit approval resulting from this transaction.

 

For mortgage accounts, the value provided in expressed in thousands.  For other, it is expressed as hundreds.

 

For example:  If seeking approval for a mortgage of $210,000, the input would be set to "210".  If $85,000 then the input would be "085".  If value is less than $1,000, the input would be zeroes "000".

 

If approving an auto loan of $15,600, for example, the input would be "156".  If seeking $2,508", the input would be "025".
