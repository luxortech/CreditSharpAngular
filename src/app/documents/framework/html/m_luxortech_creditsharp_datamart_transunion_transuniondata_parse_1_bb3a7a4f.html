TransUnionData.Parse Method

 

Namespace: LuxorTech.CreditSharp.Datamart.TransUnion

    

 

The Parse method is used to import a native credit bureau image and organize the contents into collections of segments.  Once imported, the native bureau values can be inspected directly through the exposed data classes (TucInquiry, TucScoreModel, and so on) or the resulting internal references can be passed as a single input to a CommonData object and transformed into a common, normalized view.

 

The Parse method automatically detects what layout and what version of the TransUnion response data has been provided.  The CreditSharp Framework currently can recognize and process the following inputs:

 

•Fixed Format Response layout, Version 4.0

•TUXML layout, Version 2.19 (and earlier)

 

The Parse method will also accept any print ready layouts, inserting the required spacing and pagination controls to make the output presentable.  This is also true for any of the 'combo' style images where both a machine readable layout and a print ready layout are returned in the same response.  The print ready portions of a response can be retrieved from the TransUnionData's Report property after the parse operation is complete.

 

When attempting to process two bureau inputs each must contain data on only one consumer.  An error will be thrown if either input contains a joint response.  It is possible to process different types of bureau responses when providing two inputs; for instance, one can be in the FFR format and the other might be XML.

Example 1

 

// This example demonstrates using a single BureauData base class reference

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoCommon

  {

      public string GetXmlOutput(string rawCreditInput, BureauIds bureauId)

      {

          BureauData bureauData = null;

          switch (bureauId)

          {

              case BureauIds.Equifax:

                  bureauData = new EquifaxData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.Experian:

                  bureauData = new ExperianData(new BureauOptions(licenseKey));

                  break;

              case BureauIds.TransUnion:

                  bureauData = new TransUnionData(new BureauOptions(licenseKey));

                  break;

          }

          bureauData.Parse(rawCreditInput);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(bureauData);

 

          FormatOptions options = new FormatOptions(FormatLayout.Xml);

          options.Xml.Partitions = Partitioning.IdMatch;

          FormatData formatData = new FormatData(options);

 

          return formatData.Generate(commonData);

      }

   }

}

 

Example 2

 

// This example demonstrates using the bureau specific data references

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoCommon

  {

      public string GetXmlOutput(string rawCreditInput, BureauIds bureauId)

      {

          string xmlOutput = string.Empty;

          CommonData commonData = null;

 

          switch (bureauId)

          {

              case BureauIds.Equifax:

                  commonData = DoEquifax(rawCreditInput);

                  break;

              case BureauIds.Experian:

                  commonData = DoExperian(rawCreditInput);

                  break;

              case BureauIds.TransUnion:

                  commonData = DoTransUnion(rawCreditInput);

                  break;

          }

 

          FormatOptions options = new FormatOptions(FormatLayout.Xml);

          options.Xml.Partitions = Partitioning.IdMatch;

          FormatData formatData = new FormatData(options);

 

          return formatData.Generate(commonData);

      }

 

      public CommonData DoEquifax(string rawData)

      {

          EquifaxData equifaxData = new EquifaxData(new BureauOptions(licenseKey));

          equifaxData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(equifaxData);

 

          return commonData;

      }

 

      public CommonData DoExperian(string rawData)

      {

          ExperianData experianData = new ExperianData(new BureauOptions(licenseKey));

          experianData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(experianData);

 

          return commonData;

      }

 

      public CommonData DoTransUnion(string rawData)

      {

          TransUnionData transUnionData = new TransUnionData(new BureauOptions(licenseKey));

          transUnionData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(transUnionData);

 

          return commonData;

      }

   }

}
