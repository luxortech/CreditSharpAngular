DefineGroups Method

 

Namespace: LuxorTech.CreditSharp.Format

    

 

The DefineGroups method is used to select an existing set of grouping criteria, or, to initialize the section in preparation to supply customized criteria.

 

The list of existing group criteria can be found in each section specific topic page, located within the Derived Classes chapter.  For instance, visit the FormatSectionTradeline topic to review the predefined sort and group definitions that can be used in formatting Tradeline displays.

 

 

Syntax

public void DefineGroups (string name)

 

Parameters:

 

name

Type: string

The name of the definition being selected, or, an indication that a customized sort definition is to be used.

 

Valid inputs:

 

name = "System"

The predefined CreditSharp group criteria for address members.  This is the default setting.

Note: By default, there is no grouping criteria used in Address sections.

 

name = "Custom"

A user defined set of sort criteria is to be supplied.  This will clear out any existing sort definitions for the current section.

Exceptions

Exception class

Status

FormatDataException

FormatDataStatus.InvalidOperation

Example of customizing group criteria.

 

using System;

 

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Datamart.Merge;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoFormat

  {

      public string CreateText(MergeData input, AnalyzeData aggregates)

      {

          FormatOptions options = new FormatOptions(FormatLayout.Text);

          options.Text.Partitions = Partitioning.Report;

          options.Text.ReportStyle = TextStyle.Lending01;

          options.Text.ShowSummary = true;

 

          //

          // Obtain a reference for the configuration of an address section in the text layout

          //

          FormatSectionAddress addressSection = (FormatSectionAddress) options.Text.Section("Address");

          addressSection.Verbose = true;

 

          // Prepare the section for accepting a customized setting

          addressSection.DefineGroups("CUSTOM");

 

          FormatGroupAddress addressGroup = addressSection.AddGroup("Current");

          //

          // Establish the selection criteria for the first group, using a lambda expression

          //

          addressGroup.Selector = (x => x.IsCurrentAddress == true);

          addressGroup.InitialHeader = "Current address" + Environment.NewLine +

                                        "---------------";

          //

          // Establish the selection criteria for the second group, using a lambda expression

          //

          // Best practice is to ensure that selections can only appear in one group

          //

          addressGroup = addressSection.AddGroup("Former");

          addressGroup.Selector = (x => x.IsCurrentAddress == false);

          addressGroup.InitialHeader = "Former address" + Environment.NewLine +

                                        "--------------";

          // Now, set up a custom sort order

          addressSection.DefineSort("CUSTOM");

          //

          // Use the DATE helper function to simplify date comparisons and invert the value (multiply by

          // negative one) so that items are returned in descending order (most recent dates first).

          //

          addressSection.AddSort(x => x.DATE(ComAddressIds.FirstReported) * -1);

          addressSection.AddSort(x => x.DATE(ComAddressIds.LastReported) * -1);

 

          FormatData format = new FormatData(options);

          return format.Generate(input, aggregates);

      }

   }

}

 

Example for choosing existing group criteria.

using System;

 

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart.Merge;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoFormat

  {

      public string CreateHtml(MergeData input, AnalyzeData aggregates)

      {

          FormatOptions options = new FormatOptions(FormatLayout.Html);

          options.Html.Partitions = Partitioning.Report;

          options.Html.ReportStyle = HtmlStyle.Common01;

          options.Html.TradelineStyle = HtmlTradelineStyle.Tradeline04;

 

          //

          // Obtain a reference for the configuration of a tradeline section in the Html layout

          //

          FormatSectionTradeline tradeSection = (FormatSectionTradeline)options.Html.Section("Tradeline");

 

          // Specify the desired predefined group configuration

          tradeSection.DefineGroups("OpenClosed");

 

          FormatData format = new FormatData(options);

          return format.Generate(input, aggregates);

      }

  }
