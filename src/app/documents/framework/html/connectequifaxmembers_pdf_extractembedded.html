Extracting an embedded PDF image from a JSON response

 

    

 

There are two methods for obtaining a PDF image for display purposes.  The JSON data can be referenced directly using a JSON parsing tool or the data can be presented to a CreditSharp object for processing.  Of the two methods, using CreditSharp to extract the PDF file is perhaps the easiest.

 

Note that the content embedded within the pdfImage1 (and pdfImage2, if a Joint report was requested) element is converted from its binary form and preserved using base64 encoding.  In order to subsequently display the actual PDF data this content must be first converted back to its binary form.  When using the CreditSharp Image() method, as shown in the first sample procedure, the base64 encoded is converted for you.

 

In the second procedure sample, where the JSON is parsed natively, this conversion must be done using the appropriate a .NET utility.

 

In either case, once the binary PDF data is in hand (as a byte array), the contents can be passed to a library extension capable of displaying a PDF file from a data stream or written to a system file that can then be opened in a PDF viewer.

 

namespace EquifaxPDFExtract

{

    using System;

    using System.IO;

    using System.Text.Json;

 

    using LuxorTech.CreditSharp.Datamart;

    using LuxorTech.CreditSharp.Datamart.Equifax;

 

    public class ExtractSamples

    {

        public void ExtractUsingCreditSharp(string licenseKey, string jsonContent)

        {

            EquifaxData equifaxData = new EquifaxData(new BureauOptions(licenseKey));

            equifaxData.Parse(jsonContent);

 

            byte[] pdfData;

 

            //

            // obtain the unencoded pdf content using the CreditSharp Image() method

            //

            pdfData = equifaxData.Image(false);

            if (pdfData.GetUpperBound(0) > 0)

                WritePDF(pdfData, @"C:\MyOutputFolder\borrowerReport.pdf");

 

            //

            // obtain the unencoded pdf content using the CreditSharp Image() method

            //

            // pass boolean parameter as true to retrieve the co-applicant report, if any

            //

            pdfData = equifaxData.Image(true);

            if (pdfData.GetUpperBound(0) > 0)

                WritePDF(pdfData, @"C:\MyOutputFolder\coborrowerReport.pdf");

        }

 

        public void ExtractUsingJsonParser(string jsonContent)

        {

            byte[] pdfData;

 

            using (JsonDocument jsonData = JsonDocument.Parse(jsonContent))

            {

                JsonElement element;

                if (jsonData.RootElement.TryGetProperty("pdfImage1", out element) == true)

                {

                    //

                    // convert embedded string from base64 encoding to a byte array

                    //

                    pdfData = Convert.FromBase64String(element.GetString());

                    WritePDF(pdfData, @"C:\MyOutputFolder\borrowerReport.pdf");

                }

 

                if (jsonData.RootElement.TryGetProperty("pdfImage2", out element) == true)

                {

                    //

                    // convert embedded string from base64 encoding to a byte array

                    //

                    pdfData = Convert.FromBase64String(element.GetString());

                    WritePDF(pdfData, @"C:\MyOutputFolder\coborrowerReport.pdf");

                }

            }

        }

 

        public void WritePDF(byte[] pdfContent, string filename)

        {

            File.WriteAllBytes(filename, pdfContent);

        }

    }

}
