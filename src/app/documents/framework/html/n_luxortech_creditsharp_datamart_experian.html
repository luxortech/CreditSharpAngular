LuxorTech.CreditSharp.Datamart.Experian Namespace

 

    

 

The Experian namespace presents the classes and enumerations that together allow for raw credit bureau responses to be imported and reviewed.  Once imported, the ExperianData object can be passed on as input to the Normalize method of the CommonData class to transform the information to a common set of definitions.

 

A typical programming pattern

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoImport

  {

      public string GetHtmlReport(string rawCreditInput, BureauIds bureauId)

      {

          // This example demonstrates using a bureau specific data references

          string xmlOutput = string.Empty;

          CommonData commonData = null;

 

          switch (bureauId)

           

              case BureauIds.Equifax:

                  commonData = DoEquifax(rawCreditInput);

                  break;

              case BureauIds.Experian:

                  commonData = DoExperian(rawCreditInput);

                  break;

              case BureauIds.TransUnion:

                  commonData = DoTransUnion(rawCreditInput);

                  break;

          }

 

          FormatOptions options = new FormatOptions(FormatLayout.Html);

          options.Html.ReportStyle = HtmlStyle.Common01;

          options.Html.TradelineStyle = HtmlTradelineStyle.Tradeline04;

          FormatData formatData = new FormatData(options);

 

          return formatData.Generate(commonData);

      }

 

      public CommonData DoEquifax(string rawData)

      {

          EquifaxData equifaxData = new EquifaxData(new BureauOptions(licenseKey));

          equifaxData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(equifaxData);

 

          return commonData;

      }

 

      public CommonData DoExperian(string rawData)

      {

          ExperianData experianData = new ExperianData(new BureauOptions(licenseKey));

          experianData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(experianData);

 

          return commonData;

      }

 

      public CommonData DoTransUnion(string rawData)

      {

          TransUnionData transUnionData = new TransUnionData(new BureauOptions(licenseKey));

          transUnionData.Parse(rawData);

 

          CommonData commonData = new CommonData(new CommonOptions());

          commonData.Normalize(transUnionData);

 

          return commonData;

      }

   }

}
