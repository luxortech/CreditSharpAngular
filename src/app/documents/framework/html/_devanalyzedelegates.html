Analyze Delegates

 

Namespace: LuxorTech.CreditSharp.Analyze

    

 

Delegates are used to create custom definitions for aggregating credit data values. The predicates define which occurrences of a given credit segment will be measured and solutions define the computation that will be performed on the selected occurrences.

 

Both predicates and solutions can be defined using inline lambda expressions or by providing a reference to a compiled function (or method) to be invoked.  There are specific delegates designed for use with each of the credit data objects that can be analyzed.

 

Using compiled functions for the predicate and solution delegates, rather than inline expressions, will yield faster execution speeds.  However, the speed gains are only noticeable when a large number of custom attributes are being employed.

 

Lambda Implementation

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Datamart.Common;

 

namespace HelpDemos

{

  public class PeformAnalysis

   {

      public AnalyzeData Execute(CommonData common)

       {

          AnalyzeOptions options = new AnalyzeOptions();

          options.IgnoreLostStolen = true;   // Do not consider trades reported lost or stolen

          options.IgnoreDisputed = true;     // Do not consider trades in dispute

 

                AnalysisBundle bundle = new AnalysisBundle(AnalysisId.CustomEvaluation, Partitioning.IdMatch, options, "MyCustomAnalysis");

 

                //

                // Statements can be defined using separate declarations of each lambda expression  

                //

                TradelinePredicate nbrBankCard = (trade, options) => trade.IsBankCard;

                TradelinePredicate nbrCreditLine = (trade, options) => trade.IsCreditLine;

 

                TradelinePredicate isOpenTrl = (trade, options) => trade.IsOpen && !trade.IsCollection && !trade.EMPTY(ComTradelineIds.DateOpen);

                TradelineSolution mosSinceOpen = (trade, options) => trade.AGE(ComTradelineIds.DateOpen);

 

                TradelinePredicate isTrl = (trade, options) => !trade.IsCollection;

 

                // Then, as each evaluation is created, a reference to the distinct delegate expression is provided

                bundle.AddEvaluation(new EvaluateTradeline("NbrBankCard", AnalysisOperation.Count, ComTradelineIds.IdCount, nbrBankCard));

                bundle.AddEvaluation(new EvaluateTradeline("NbrCreditLine", AnalysisOperation.Count, ComTradelineIds.IdCount, nbrCreditLine));

                bundle.AddEvaluation(new EvaluateTradeline("MonthsSinceNewestOpen", AnalysisOperation.Minimum, ComTradelineIds.IdCount, isOpenTrl, mosSinceOpen));

                bundle.AddEvaluation(new EvaluateTradeline("MonthsSinceOldestOpen", AnalysisOperation.Maximum, ComTradelineIds.IdCount, isOpenTrl, mosSinceOpen));

                bundle.AddEvaluation(new EvaluateTradeline("NewestOpenDate", AnalysisOperation.Minimum, ComTradelineIds.DateOpen, isTrl));

                bundle.AddEvaluation(new EvaluateTradeline("OldestOpenDate", AnalysisOperation.Maximum, ComTradelineIds.DateOpen, isTrl));

 

                //

                // Alternatively, statements can be supplied as inline lambda expressions directly when creating an evaluation  

                //

                bundle.AddEvaluation(new EvaluateTradeline("NbrLostStolen", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsLostStolen));

                bundle.AddEvaluation(new EvaluateTradeline("NbrDisputed", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsDisputed));

                bundle.AddEvaluation(new EvaluateTradeline("NbrTrades", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => !trade.IsCollection));

                bundle.AddEvaluation(new EvaluateTradeline("NbrCollections", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsCollection));

                bundle.AddEvaluation(new EvaluateTradeline("NbrOpenTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => !trade.IsCollection && trade.ClosedIndicator == "0"));

                bundle.AddEvaluation(new EvaluateTradeline("NbrOpenCol", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsCollection && trade.ClosedIndicator == "0"));

 

          List<AnalysisBundle> bundles = new List<AnalysisBundle>();

          bundles.Add(bundle);

          AnalyzeData analyze = new AnalyzeData(bundles);

 

          analyze.Evaluate(common);

          return analyze;

       }

   }

}

 

Function Implementation

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Datamart.Common;

 

namespace HelpDemos

{

  public class PeformAnalysis

   {

      public AnalyzeData Execute(CommonData common)

       {

          AnalyzeOptions options = new AnalyzeOptions();

          options.IgnoreLostStolen = true;   // Do not consider trades reported lost or stolen

          options.IgnoreDisputed = true;     // Do not consider trades in dispute

 

                AnalysisBundle bundle = new AnalysisBundle(AnalysisId.CustomEvaluation, Partitioning.IdMatch, options, "MyCustomAnalysis");

 

                //

                // Each statement which defines a custom attribute is constructed using a compiled static function

                //

                bundle.AddEvaluation(new EvaluateTradeline("NbrBankCard", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrBankCard));

                bundle.AddEvaluation(new EvaluateTradeline("NbrCreditLine", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrCreditLine));

                bundle.AddEvaluation(new EvaluateTradeline("NbrLostStolen", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrLostStolen));

                bundle.AddEvaluation(new EvaluateTradeline("NbrDisputed", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrDisputed));

                bundle.AddEvaluation(new EvaluateTradeline("NbrTrades", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrTrades));

                bundle.AddEvaluation(new EvaluateTradeline("NbrCollections", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrCollections));

                bundle.AddEvaluation(new EvaluateTradeline("NbrOpenTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrOpenTrl));

                bundle.AddEvaluation(new EvaluateTradeline("NbrOpenCol", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)NbrOpenCol));

                // When both a Predicate and Solution are being used, there is not need to cast each reference to the appropriate delegate type. That is inferred

                // since both parameters are being specified.

                bundle.AddEvaluation(new EvaluateTradeline("MonthsSinceNewestOpen", AnalysisOperation.Minimum, ComTradelineIds.IdCount, IsOpenTrade, MonthsSinceOpen));

                bundle.AddEvaluation(new EvaluateTradeline("MonthsSinceOldestOpen", AnalysisOperation.Maximum, ComTradelineIds.IdCount, IsOpenTrade, MonthsSinceOpen));

                bundle.AddEvaluation(new EvaluateTradeline("NewestOpenDate", AnalysisOperation.Minimum, ComTradelineIds.DateOpen, (TradelinePredicate)IsTrade));

                bundle.AddEvaluation(new EvaluateTradeline("OldestOpenDate", AnalysisOperation.Maximum, ComTradelineIds.DateOpen, (TradelinePredicate)IsTrade));

 

          List<AnalysisBundle> bundles = new List<AnalysisBundle>();

          bundles.Add(bundle);

          AnalyzeData analyze = new AnalyzeData(bundles);

 

          analyze.Evaluate(common);

          return analyze;

       }

 

            private static bool NbrBankCard(ComTradeline trade, AnalyzeOptions options) { return (trade.IsBankCard); }

            private static bool NbrCreditLine(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCreditLine); }

            private static bool NbrLostStolen(ComTradeline trade, AnalyzeOptions options) { return (trade.IsLostStolen); }

            private static bool NbrDisputed(ComTradeline trade, AnalyzeOptions options) { return (trade.IsDisputed); }

            private static bool NbrTrades(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection); }

            private static bool NbrCollections(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCollection); }

            private static bool NbrOpenTrl(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection && trade.ClosedIndicator == "0"); }

            private static bool NbrOpenCol(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCollection && trade.ClosedIndicator == "0"); }

 

            private static bool IsOpenTrade(ComTradeline trade, AnalyzeOptions options) { return (trade.IsOpen && !trade.IsCollection); }

            private static double MonthsSinceOpen(ComTradeline trade, AnalyzeOptions options) { return (trade.AGE(ComTradelineIds.DateOpen)); }

            private static bool IsTrade(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection); }

   }

}
