Retrieving a PDF image from Equifax

 

    

 

Once a credit inquiry has been submitted to the bureau with the PDF report option enabled, Equifax will generate a PDF image for the consumer(s) and store that file on their server for up to two hours following the initial request.  Then, as part of the JSON credit bureau response, a link to the PDF report(s) is provided so that the file can be downloaded using a follow up request.

 

If you have elected to not have CreditSharp automatically embed the PDF report within the JSON credit data envelope, then the download operation can be performed separately.  This follow up request is in itself a complete credit bureau transaction in which Equifax credentials are needed to authenticate the request before the PDF report can be returned.  This separate transaction can be performed outside of the CreditSharp domain, but it will most likely be most convenient to leverage the CreditSharp framework to retrieve the report(s).

 

The programming sample below demonstrates how to use CreditSharp to perform the separate download transaction.  It includes a small function showing a quick and easy method for getting the PDF link(s) from the original JSON credit data response:

 

 

namespace EquifaxPDFRetrieve

{

    using System.Text.Json;

 

    using LuxorTech.CreditSharp.Core;

    using LuxorTech.CreditSharp.Connect;

 

    public class RetrieveEquifaxPdf

    {

        public string MyAuthorizationId;

        public string MyAuthorizationSecret;

 

        public byte[] DownLoadData(string licenseKey, string equifaxLink)

        {

            BureauConnection connection = new BureauConnection(BureauIds.Equifax);

            connection.HttpSettings.ServiceURL = @"https://api.uat.equifax.com/v2/oauth/token";

            connection.HttpSettings.ServiceName = "api";

            connection.HttpSettings.AuthorizationId = this.MyAuthorizationId;

            connection.HttpSettings.AuthorizationSecret = this.MyAuthorizationSecret;

 

            //

            // Create the options reference and provide the PDF URL in the PdfLink property

            //

            EfxInquiryOptions options = new EfxInquiryOptions(licenseKey)

            {

                PdfLink = equifaxLink

            };

 

            EquifaxRequest request = new EquifaxRequest(options);

 

            //

            // Set up the bare-minimum request parameters

            //

            request.ResponseVersion = EfxVersionNumber.Version10;

            request.ProductCode = EfxProductCode.CreditReport;

            request.TtyResponse = EfxTtyResponse.PdfImage;

 

            //

            // Send off the request to the Equifax server.

            //

            // Note that normal Applicant parameter is passed here as a null value.

            //

            string base64Data = request.RequestProduct(connection, null);

 

            //

            // If you will be directly displaying or saving the data to a physical PDF file, then return the binary

            // form of the PDF report by reversing the base64 encoding.

            //

            // If you will be saving in a database you may still want the binary form, since it will be smaller, and

            // store that content in a BLOB database column.  Or, you can elect to store the data in its ASCII form

            // by simply returning the base64Data and saving the reversal of the encoding for a later time.

            //

            return System.Convert.FromBase64String(base64Data);

        }

 

        //

        // Get the Equifax link for the PDF reports belonging to each borrower 

        //

        public void GetURLs(string equifaxJsonData, out string borrowerLink, out string coborrowerLink)

        {

            borrowerLink = string.Empty;

            coborrowerLink = string.Empty;

 

            bool first = true;

            using (JsonDocument jsonData = JsonDocument.Parse(equifaxJsonData))

            {

                JsonElement element;

                if (jsonData.RootElement.TryGetProperty("links", out element) == true)

                {

                    JsonElement.ArrayEnumerator links = element.EnumerateArray();

                    while (links.MoveNext())

                    {

                        JsonElement item = links.Current;

                        if (first == true)

                        {

                            borrowerLink = item.GetProperty("href").GetString();

                            first = false;

                        }

                        else

                            coborrowerLink = item.GetProperty("href").GetString();

                    }

                }

            }

        }

    }

}
