Summary Data Reference

 

    

 

The following attributes are computed when the prebuilt AnalysisId.SummaryData01 bundle is evaluated.  These computed results are the basis for the Summary data sections in the various Format outputs.

 

A brief programming example

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Datamart.Common;

 

namespace HelpDemos

{

  public class PeformAnalysis

   {

      public AnalyzeData Execute(CommonData common)

       {

          AnalyzeOptions options = new AnalyzeOptions();

           options.IgnoreLostStolen = true;   // Do not consider trades reported lost or stolen

           options.IgnoreDisputed = true;     // Do not consider trades in dispute

 

          List<AnalysisBundle> bundles = new List<AnalysisBundle>();

          bundles.Add(new AnalysisBundle(AnalysisId.SummaryData01, Partitioning.IdMatch, options));

           bundles.Add(new AnalysisBundle(AnalysisId.AttibuteFactory01, Partitioning.Report, options));

 

          AnalyzeData analyze = new AnalyzeData(bundles);

 

           analyze.Evaluate(common);

          return analyze;

       }

   }

}
