Streamlining Code

 

    

 

In many places throughout the CreditSharp framework, there are bureau specific objects which each share a common base class.  This allows for some measure of code reduction by developing functions which operate on the base class rather than the specific bureau objects.

 

For example, the BureauRequest type can be used to consolidate operations within the Connect namespace.  In the following sample, the bureau specific work is relegated to subroutines while the primary execution logic is carried out in the top level function:

 

A Connect namespace example

using System;

using System.Diagnostics;

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Connect;

 

namespace HelpDemos

{

  public class OrderBureau

  {

      public string SubmitRequest(BureauIds bureauId, Applicant primary, Applicant secondary)

      {

          BureauConnection connection = null;

          BureauRequest request = null;

          string output = string.Empty;

 

          try

          {

              switch (bureauId)

              {

                  case BureauIds.Equifax:

                      ConfigureEquifax(out connection, out request);

                      break;

                  case BureauIds.Experian:

                      ConfigureExperian(out connection, out request);

                      break;

                  case BureauIds.TransUnion:

                      ConfigureTransUnion(out connection, out request);

                      break;

              }

 

              if (secondary != null)

                  request.IsJointRequest = true;

          }

 

          catch (CreditSharpException ex)

          {

              Debug.WriteLine("An error occurred constructing inquiry for {0}",

                                primary.Identity.FirstName + " " + primary.Identity.LastName);

              Debug.WriteLine(ex.Message);

              output = "Request failed.";

          }

 

          try

          {

              output = request.RequestProduct(connection, primary, secondary);

          }

 

          catch (CreditSharpException ex)

          {

              CreditSharpStatus status = new CreditSharpStatus(ex.EncodedStatus);

              if (status.Domain == ExceptionDomain.CreditBureau)

                  Debug.WriteLine("A credit bureau error occurred during communications.");

              else

                  Debug.WriteLine("An error occurred when attempting bureau communications.");

              Debug.WriteLine(ex.Message);

              output = "Request failed.";

          }

 

          return output;

      }

 

      internal void ConfigureEquifax(out BureauConnection connect, out BureauRequest request)

      {

          connect = new BureauConnection(BureauIds.Equifax);

          connect.HttpSettings.ServiceName = "ACROTEST";

          connect.HttpSettings.ServiceURL = "https://transport5.ec.equifax.com/servlet/stspost";

          connect.HttpSettings.WebUserId = "myCompanyId";

          connect.HttpSettings.WebPassword = "letMeIn";

 

          EfxInquiryOptions options = new EfxInquiryOptions(licenseKey);

          options.ConsumerReferral = true;

          options.MilitaryBorrowerCheck = true;

 

          EquifaxRequest equifax = new EquifaxRequest(options);

          equifax.PermissiblePurposeCode = "01";

          equifax.ProductCode = EfxProductCode.CreditReport;

          equifax.ResponseVersion = EfxVersionNumber.Version50;

          equifax.SubscriberCode = "222XX11111";

          equifax.SubscriberPassword = "ABC";

 

          request = equifax;

      }

 

      internal void ConfigureExperian(out BureauConnection connect, out BureauRequest request)

      {

          connect = new BureauConnection(BureauIds.Experian);

          connect.HttpSettings.ServiceName = "STAR";

          connect.HttpSettings.ServiceURL = "http://www.experian.com/lookupServlet1?lookupServiceName=AccessPoint&lookupServiceVersion=1.0&serviceName=NetConnectDemo&serviceVersion=2.0&responseType=text/plain";

          connect.HttpSettings.WebUserId = "myCompanyId";

          connect.HttpSettings.WebPassword = "notPassword";

 

          ExperianRequest experian = new ExperianRequest(new ExpInquiryOptions(licenseKey));

          experian.PermissiblePurposeCode = "01";

          experian.ProductCode = ExpProductCode.CreditReport;

          experian.ResponseVersion = ExpVersionNumber.Version70;

          experian.SubscriberPreamble = "TEST";

          experian.SubscriberCode = "1234567";

          experian.SubscriberPassword = "ABC";

          experian.XmlResponse = ExpXmlResponse.Standard;

 

          request = experian;

      }

 

      internal void ConfigureTransUnion(out BureauConnection connect, out BureauRequest request)

      {

          connect = new BureauConnection(BureauIds.TransUnion);

          connect.HttpSettings.ServiceURL = "https://netaccess-test.transunion.com";

          connect.HttpSettings.WebUserId = "myCompanyId";

          connect.HttpSettings.WebPassword = "notPassword";

 

          connect.HttpSettings.CertificateStore = "Trusted Root";

          connect.HttpSettings.CertificateLocation = "CurrentUser";

          connect.HttpSettings.CertificateName = "TUNA Cert";

 

          TucInquiryOptions options = new TucInquiryOptions(licenseKey);

          options.RequestRegion = TucRequestRegion.ProductionSystem;

 

          TransUnionRequest transunion = new TransUnionRequest(options);

          transunion.PermissiblePurposeCode = "AU";

          transunion.ProductCode = TucProductCode.CreditReport;

          transunion.ResponseVersion = TucVersionNumber.Version40;

          transunion.SubscriberCode = "1234567";

          transunion.SubscriberPassword = "A999";

          transunion.SubscriberMarket = "06";

          transunion.SubscriberSubMarket = "22";

          transunion.TtyResponse = TucTtyResponse.ContinuousSheet;

 

          request = transunion;

      }

   }

}

 

In similar fashion, the BureauData type can be used to consolidate operations within the Datamart namespace.  Any bureau specific is performed using the appropriate Datamart object, which is then immediately cast as a BureauData type for further processing.

 

A Datamart namespace example

 

using System;

using System.Diagnostics;

 

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart;

using LuxorTech.CreditSharp.Datamart.Equifax;

using LuxorTech.CreditSharp.Datamart.Experian;

using LuxorTech.CreditSharp.Datamart.TransUnion;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Format;

 

namespace HelpDemos

{

  public class DemoParseStreamlined

  {

      public string GetHtmlReport(BureauIds bureauId, Applicant primary, Applicant secondary)

      {

          string output = string.Empty;

          OrderBureau orderBureau = new OrderBureau();

          string creditData = orderBureau.SubmitRequest(bureauId, primary, secondary);

 

          if (creditData == "Request failed.")

              return "Request failed.";

 

          BureauData bureauData = null;

          try

          {

              switch (bureauId)

              {

                  case BureauIds.Equifax:

                      bureauData = new EquifaxData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.Experian:

                      bureauData = new ExperianData(new BureauOptions(licenseKey));

                      break;

                  case BureauIds.TransUnion:

                      bureauData = new TransUnionData(new BureauOptions(licenseKey));

                      break;

              }

 

              bureauData.Parse(creditData);

 

              CommonData commonData = new CommonData(new CommonOptions());

              commonData.Normalize(bureauData);

 

              FormatData formatData = new FormatData(new FormatOptions(FormatLayout.Html));

              output = formatData.Generate(commonData);

          }

 

          catch (CreditSharpException ex)

          {

              if (ex is BureauDataException)

              {

                  Debug.WriteLine("An error occurred importing bureau data on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is NormalizeDataException)

              {

                  Debug.WriteLine("An error occurred normalizing bureau data on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is FormatDataException)

              {

                  Debug.WriteLine("An error occurred generating report on {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              else if (ex is ParameterException)

              {

                  Debug.WriteLine("An input paremter is incorrect.");

              }

              else

              {

                  Debug.WriteLine("An error while processing the request for {0}",

                                      primary.Identity.FirstName + " " + primary.Identity.LastName);

              }

              Debug.WriteLine(ex.Message);

              output = "Request failed.";

          }

 

          return output;

      }

   }

}
