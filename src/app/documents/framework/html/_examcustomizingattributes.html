Customizing Analyze Attributes

 

    

 

The following example presents a complete working solution illustrating the creation of customized attribute definitions for use within the Analyze namespace.

 

The coding samples highlight implementing the customized definitions using either inline lambda expressions or by referencing compiled functions/methods.  This topic also presents a representative samples of how these custom attributes would appear in an XML or JSON output.

 

Custom attribute creation example

using System.Collections.Generic;

 

using LuxorTech.CreditSharp.Analyze;

using LuxorTech.CreditSharp.Core;

using LuxorTech.CreditSharp.Datamart.Common;

using LuxorTech.CreditSharp.Datamart.Merge;

 

namespace AnalyzeTestBed

{

  public class AnalyzeDemo

  {

      public AnalyzeData PerformAnalysis(CommonData commonData, bool executionMethod, AnalyzeOptions options)

      {

          return this.Execute(commonData, null, executionMethod, options);

      }

 

      public AnalyzeData PerformAnalysis(MergeData mergeData, bool executionMethod, AnalyzeOptions options)

      {

          return this.Execute(null, mergeData, executionMethod, options);

      }

 

      private AnalyzeData Execute(CommonData commonData, MergeData mergeData, bool executionMethod, AnalyzeOptions options)

      {

          List<AnalysisBundle> bundles = new List<AnalysisBundle>();

          AnalysisBundle bundle = new AnalysisBundle(AnalysisId.CustomEvaluation, Partitioning.IdMatch, options, "MyCustomAnalysis");

 

          options.IgnoreDisputed = !options.IgnoreDisputed;

 

          if (executionMethod == true)

          {

              InlinedIdentification(bundle);

              InlinedInquiry(bundle);

              InlinedPublicRecord(bundle);

              InlinedReference(bundle);

              InlinedTradeline(bundle);

          }

          else

          {

              InvokedIdentification(bundle);

              InvokedInquiry(bundle);

              InvokedPublicRecord(bundle);

              InvokedReference(bundle);

              InvokedTradeline(bundle);

          }

 

          bundles.Add(bundle);

          AnalyzeData analyzeData = new AnalyzeData(bundles);

 

          if (mergeData != null)

              analyzeData.Evaluate(mergeData);

          else

              analyzeData.Evaluate(commonData);

 

          return analyzeData;

      }

 

      /// <summary>

      /// Instructions for defining and adding custom Identification attributes using inline lambda references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InlinedIdentification(AnalysisBundle bundle)

      {

          //

          // Statements can be defined using separate declarations of each lambda expression  

          //

          IdentificationSolution nbrDaysHistory = (ident, options) => ident.AGE2(ComIdentificationIds.FirstReported);

          bundle.AddEvaluation(new EvaluateIdentification("IDT_DaysOnFile", AnalysisOperation.Compute, ComIdentificationIds.IdCount, nbrDaysHistory));

      }

 

      /// <summary>

      /// Instructions for defining and adding custom Inquiry attributes using inline lambda references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InlinedInquiry(AnalysisBundle bundle)

      {

          //

          // Statements can be defined using separate declarations of each lambda expression  

          //

          InquiryPredicate nbrAutoInqr = (inquiry, options) => inquiry.IsAuto;

          InquiryPredicate nbrAutoInqr2 = (inquiry, options) => inquiry.AGE2(ComInquiryIds.InquiryDate) < 15;

          InquirySolution daysSinceInqr = (inquiry, options) => inquiry.AGE2(ComInquiryIds.InquiryDate);

 

          // Then, as each evaluation is created, a reference to the distinct delegate expression is provided

          bundle.AddEvaluation(new EvaluateInquiry("INQ_NbrAuto", AnalysisOperation.Count, ComInquiryIds.IdCount, nbrAutoInqr));

          bundle.AddEvaluation(new EvaluateInquiry("INQ_NbrAuto2", AnalysisOperation.Count, ComInquiryIds.IdCount, nbrAutoInqr2));

          bundle.AddEvaluation(new EvaluateInquiry("INQ_DaysSinceNewest", AnalysisOperation.Minimum, ComInquiryIds.IdCount, daysSinceInqr));

          bundle.AddEvaluation(new EvaluateInquiry("INQ_DaysSinceOldest", AnalysisOperation.Maximum, ComInquiryIds.IdCount, daysSinceInqr));

 

          //

          // Alternatively, statements can be supplied as inline lambda expressions directly when creating an evaluation  

          //

          bundle.AddEvaluation(new EvaluateInquiry("INQ_Count", AnalysisOperation.Count, ComInquiryIds.IdCount, (inquiry, options) => true));

          bundle.AddEvaluation(new EvaluateInquiry("INQ_Count2", AnalysisOperation.Count, ComInquiryIds.IdCount, (inquiry, options) => inquiry.AGE2(ComInquiryIds.InquiryDate) < 15));

      }

      }

 

      /// <summary>

      /// Instructions for defining and adding custom Reference (Compute) attributes using inline lambda references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InlinedReference(AnalysisBundle bundle)

      {

          //

          // This example illustrates how to use internal attribute references to compute a value based upon other aggregates.

          // Here, a revolving utilization percentage is computed.

          //

 

          //

          // First define global references to be used for computing Revolving Utilization. Make sure we only include tradelines that are:

          //      A) Revolving, and

          //      B) have a valid CreditLimit or HighCredit

          //

          // The special '@' character indicates the attribute value is to be used for internal calculations only and will not be included in any outputs.

          // Reference attributes can include any aggregated value, not just these internal only values.

          //

          TradelinePredicate isRevWithAmts = (trade, options) => trade.IsRevolving && (!trade.EMPTY(ComTradelineIds.CreditLimitAmount) || !trade.EMPTY(ComTradelineIds.HighCreditAmount));

          TradelineSolution utlizationAmt = (trade, options) => trade.EMPTY(ComTradelineIds.CreditLimitAmount) ? trade.AMT(ComTradelineIds.HighCreditAmount) : trade.AMT(ComTradelineIds.CreditLimitAmount);

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevBalances", AnalysisOperation.Sum, ComTradelineIds.BalanceAmount, isRevWithAmts));

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevLimits", AnalysisOperation.Sum, ComTradelineIds.IdCount, isRevWithAmts, utlizationAmt));

 

          //

          // Now, construct the utilization computation using attribute references.  

          // If the Balance is zero, utilization must be zero, otherwise compute the percentage

          //

          bundle.AddEvaluation(new EvaluateReference("TRL_Utilization", AnalysisOperation.Compute,

          (results) => results.Item("@TRL_RevBalances").Quantity == 0 ? 0 :

          100 * (results.Item("@TRL_RevBalances").Quantity / results.Item("@TRL_RevLimits").Quantity)));

 

          TradelinePredicate isRev = (trade, options) => trade.IsRevolving;

          TradelinePredicate isIns = (trade, options) => trade.IsInstallment;

          TradelinePredicate isMtg = (trade, options) => trade.IsMortgage;

          TradelinePredicate isCrl = (trade, options) => trade.IsCreditLine;

          TradelinePredicate notColl = (trade, options) => !trade.IsCollection;

 

          //

          // Compute evaluations used to calculate the relative ratios to all tradelines for several account types

          //

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrRev", AnalysisOperation.Count, ComTradelineIds.IdCount, isRev));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrIns", AnalysisOperation.Count, ComTradelineIds.IdCount, isIns));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrMtg", AnalysisOperation.Count, ComTradelineIds.IdCount, isMtg));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrCrl", AnalysisOperation.Count, ComTradelineIds.IdCount, isCrl));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, notColl));

 

          // The ComputePredicate is used to ensure no attempt is made to divide by zero (if there are no trades at all)

          ComputePredicate haveTrl = (results) => results.Item("TRL_NbrTrl").Quantity > 0;

 

          ComputeSolution pctRev = (results) => results.Item("TRL_NbrRev").Quantity == 0 ? 0 :

                                   100 * (results.Item("TRL_NbrRev").Quantity / results.Item("TRL_NbrTrl").Quantity);

 

          ComputeSolution pctIns = (results) => results.Item("TRL_NbrIns").Quantity == 0 ? 0 :

                                   100 * (results.Item("TRL_NbrIns").Quantity / results.Item("TRL_NbrTrl").Quantity);

 

          ComputeSolution pctMtg = (results) => results.Item("TRL_NbrMtg").Quantity == 0 ? 0 :

                                   100 * (results.Item("TRL_NbrMtg").Quantity / results.Item("TRL_NbrTrl").Quantity);

 

          ComputeSolution pctCrl = (results) => results.Item("TRL_NbrCrl").Quantity == 0 ? 0 :

                                   100 * (results.Item("TRL_NbrCrl").Quantity / results.Item("TRL_NbrTrl").Quantity);

 

          bundle.AddEvaluation(new EvaluateReference("TRL_PctRev", AnalysisOperation.Compute, haveTrl, pctRev));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctIns", AnalysisOperation.Compute, haveTrl, pctIns));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctMtg", AnalysisOperation.Compute, haveTrl, pctMtg));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctCrl", AnalysisOperation.Compute, haveTrl, pctCrl));

      }

 

      /// <summary>

      /// Instructions for defining and adding custom Tradeline attributes using inline lambda references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InlinedTradeline(AnalysisBundle bundle)

      {

          //

          // Statements can be defined using separate declarations of each lambda expression  

          //

          TradelinePredicate nbrBankCard = (trade, options) => trade.IsBankCard;

          TradelinePredicate nbrCreditLine = (trade, options) => trade.IsCreditLine;

 

          TradelinePredicate isOpenTrl = (trade, options) => trade.IsOpen && !trade.IsCollection && !trade.EMPTY(ComTradelineIds.DateOpen);

          TradelineSolution mosSinceOpen = (trade, options) => trade.AGE(ComTradelineIds.DateOpen);

          TradelinePredicate isNotColl = (trade, options) => !trade.IsCollection;

 

          // Then, as each evaluation is created, a reference to the distinct delegate expression is provided

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrBankCard", AnalysisOperation.Count, ComTradelineIds.IdCount, nbrBankCard));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrCreditLine", AnalysisOperation.Count, ComTradelineIds.IdCount, nbrCreditLine));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_MonthsSinceNewestOpen", AnalysisOperation.Minimum, ComTradelineIds.IdCount, isOpenTrl, mosSinceOpen));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_MonthsSinceOldestOpen", AnalysisOperation.Maximum, ComTradelineIds.IdCount, isOpenTrl, mosSinceOpen));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NewestOpenDate", AnalysisOperation.Newest, ComTradelineIds.DateOpen, isNotColl));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_OldestOpenDate", AnalysisOperation.Oldest, ComTradelineIds.DateOpen, isNotColl));

 

          //

          // Alternatively, statements can be supplied as inline lambda expressions directly when creating an evaluation  

          //

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrLostStolen", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsLostStolen));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrDisputed", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsDisputed));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrTrades", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => !trade.IsCollection));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrCollections", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsCollection));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrOpenTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => !trade.IsCollection && trade.ClosedIndicator == "0"));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrOpenCol", AnalysisOperation.Count, ComTradelineIds.IdCount, (trade, options) => trade.IsCollection && trade.ClosedIndicator == "0"));

 

          bundle.AddEvaluation(new EvaluateTradeline("TRL_AverageBalance", AnalysisOperation.Average, ComTradelineIds.BalanceAmount, (trade, options) => !trade.IsCollection));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_AverageAge", AnalysisOperation.Average, ComTradelineIds.IdCount,

                              (trade, options) => trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen)));

          //

          // The same average age calculation, with added test to ensure all dates open are not blank (an extremely rare, but posible condition)

          //

          bundle.AddEvaluation(new EvaluateTradeline("TRL_AverageAge2", AnalysisOperation.Average, ComTradelineIds.IdCount,

                              (trade, options) => !trade.EMPTY(ComTradelineIds.DateOpen),

                              (trade, options) => trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen)));

      }

      /// <summary>

      /// Instructions for defining and adding custom Reference (Compute) attributes using method/function references

      /// </summary>

      /// <param name="bundle">A reference to an existing AnalysisBundle object</param>

      private void InvokedReference(AnalysisBundle bundle)

      {

          //

          // This example illustrates how to use internal attribute references to compute a value based upon other aggregates.

          // Here, a revolving utilization percentage is computed.

          //

 

          //

          // First define global references to be used for computing Revolving Utilization. Make sure we only include tradelines that are:

          //      A) Revolving, and

          //      B) have a valid CreditLimit or HighCredit

          //

          // The special '@' character indicates the attribute value is to be used for internal calculations only and will not be included in any outputs.

          // Reference attributes can include any aggregated value, not just these internal only values.

          //

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevBalances", AnalysisOperation.Sum, ComTradelineIds.BalanceAmount, (TradelinePredicate)REF_IsRevWithAmts));

          bundle.AddEvaluation(new EvaluateTradeline("@TRL_RevLimits", AnalysisOperation.Sum, ComTradelineIds.IdCount, REF_IsRevWithAmts, REF_UtilizationAmount));

          bundle.AddEvaluation(new EvaluateReference("TRL_Utilization", AnalysisOperation.Compute, REF_Utilization));

 

          //

          // Compute evaluations used to calculate the relative ratios to all tradelines for several account types

          //

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrRev", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)REF_IsRevolving));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrIns", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)REF_IsInstallment));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrMtg", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)REF_IsMortgage));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrCrl", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)REF_IsCreditLine));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrTrl", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)REF_IsNotColl));

 

          bundle.AddEvaluation(new EvaluateReference("TRL_PctRev", AnalysisOperation.Compute, REF_PctRev));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctIns", AnalysisOperation.Compute, REF_PctIns));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctMtg", AnalysisOperation.Compute, REF_PctMtg));

          bundle.AddEvaluation(new EvaluateReference("TRL_PctCrl", AnalysisOperation.Compute, REF_PctCrl));

      }

 

      private static bool REF_IsRevolving(ComTradeline trade, AnalyzeOptions options) { return trade.IsRevolving; }

      private static bool REF_IsInstallment(ComTradeline trade, AnalyzeOptions options) { return trade.IsInstallment; }

      private static bool REF_IsMortgage(ComTradeline trade, AnalyzeOptions options) { return trade.IsMortgage; }

      private static bool REF_IsCreditLine(ComTradeline trade, AnalyzeOptions options) { return trade.IsCreditLine; }

      private static bool REF_IsNotColl(ComTradeline trade, AnalyzeOptions options) { return !trade.IsCollection; }

 

      private static bool REF_IsRevWithAmts(ComTradeline trade, AnalyzeOptions options)

      {

          if (trade.IsRevolving && (!trade.EMPTY(ComTradelineIds.CreditLimitAmount) || !trade.EMPTY(ComTradelineIds.HighCreditAmount)))

              return true;

          else

              return false;

      }

 

      private static double REF_PctRev(AnalyzeResult results)

      {

          if (results.Item("TRL_NbrTrl").Quantity == 0)

              return 0;

 

          if (results.Item("TRL_NbrRev").Quantity == 0)

              return 0;

          else

              return 100 * (results.Item("TRL_NbrRev").Quantity / results.Item("TRL_NbrTrl").Quantity);

      }

 

      private static double REF_PctIns(AnalyzeResult results)

      {

          if (results.Item("TRL_NbrTrl").Quantity == 0)

              return 0;

 

          if (results.Item("TRL_NbrIns").Quantity == 0)

              return 0;

          else

              return 100 * (results.Item("TRL_NbrIns").Quantity / results.Item("TRL_NbrTrl").Quantity);

      }

 

      private static double REF_PctMtg(AnalyzeResult results)

      {

          if (results.Item("TRL_NbrTrl").Quantity == 0)

              return 0;

 

          if (results.Item("TRL_NbrMtg").Quantity == 0)

              return 0;

          else

              return 100 * (results.Item("TRL_NbrMtg").Quantity / results.Item("TRL_NbrTrl").Quantity);

      }

 

      private static double REF_PctCrl(AnalyzeResult results)

      {

          if (results.Item("TRL_NbrTrl").Quantity == 0)

              return 0;

 

          if (results.Item("TRL_NbrCrl").Quantity == 0)

              return 0;

          else

              return 100 * (results.Item("TRL_NbrCrl").Quantity / results.Item("TRL_NbrTrl").Quantity);

      }

 

      private static double REF_UtilizationAmount(ComTradeline trade, AnalyzeOptions options)

      {

          if (trade.EMPTY(ComTradelineIds.CreditLimitAmount))

              return trade.AMT(ComTradelineIds.HighCreditAmount);

          else

              return trade.AMT(ComTradelineIds.CreditLimitAmount);

      }

 

      private static double REF_Utilization(AnalyzeResult results)

      {

          if (results.Item("@TRL_RevBalances").Quantity == 0)

              return 0;

          else

              return 100 * (results.Item("@TRL_RevBalances").Quantity / results.Item("@TRL_RevLimits").Quantity);

      }

          bundle.AddEvaluation(new EvaluateTradeline("TRL_NbrOpenCol", AnalysisOperation.Count, ComTradelineIds.IdCount, (TradelinePredicate)TRL_NbrOpenCol));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_OldestOpenDate", AnalysisOperation.Oldest, ComTradelineIds.DateOpen, (TradelinePredicate)TRL_IsNotCollection));

          // since both parameters are being specified.

          bundle.AddEvaluation(new EvaluateTradeline("TRL_MonthsSinceNewestOpen", AnalysisOperation.Minimum, ComTradelineIds.IdCount, TRL_IsOpenTrade, TRL_MonthsSinceOpen));

          bundle.AddEvaluation(new EvaluateTradeline("TRL_MonthsSinceOldestOpen", AnalysisOperation.Maximum, ComTradelineIds.IdCount, TRL_IsOpenTrade, TRL_MonthsSinceOpen));

      private static double TRL_Age(ComTradeline trade, AnalyzeOptions options) { return (trade.DATE(ComTradelineIds.DateReported) - trade.DATE(ComTradelineIds.DateOpen)); }

      private static bool TRL_HaveDateOpen(ComTradeline trade, AnalyzeOptions options) { return (!trade.EMPTY(ComTradelineIds.DateOpen)); }

      private static bool TRL_NbrBankCard(ComTradeline trade, AnalyzeOptions options) { return (trade.IsBankCard); }

      private static bool TRL_NbrCreditLine(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCreditLine); }

      private static bool TRL_NbrLostStolen(ComTradeline trade, AnalyzeOptions options) { return (trade.IsLostStolen); }

      private static bool TRL_NbrDisputed(ComTradeline trade, AnalyzeOptions options) { return (trade.IsDisputed); }

      private static bool TRL_NbrTrades(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection); }

      private static bool TRL_NbrCollections(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCollection); }

      private static bool TRL_NbrOpenTrl(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection && trade.ClosedIndicator == "0"); }

      private static bool TRL_NbrOpenCol(ComTradeline trade, AnalyzeOptions options) { return (trade.IsCollection && trade.ClosedIndicator == "0"); }

 

      private static bool TRL_IsOpenTrade(ComTradeline trade, AnalyzeOptions options) { return (trade.IsOpen && !trade.IsCollection); }

      private static double TRL_MonthsSinceOpen(ComTradeline trade, AnalyzeOptions options) { return (trade.AGE(ComTradelineIds.DateOpen)); }

      private static bool TRL_IsNotCollection(ComTradeline trade, AnalyzeOptions options) { return (!trade.IsCollection); }
